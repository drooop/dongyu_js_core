### Ran Playwright code
```js
await (async (page) => {
  const user = "@drop:localhost";
  const pass = "TQcps@123";
  let uiEventReq = 0;
  page.on('request', (req) => { if (req.url().includes('/ui_event')) uiEventReq += 1; });

  const loginBtn = page.getByRole('button', { name: 'Login with Matrix' });
  if (await loginBtn.count()) {
    await page.getByRole('textbox', { name: '@user:localhost' }).fill(user);
    await page.getByRole('textbox', { name: 'Password' }).fill(pass);
    await loginBtn.click();
    await page.waitForTimeout(1000);
  }

  await page.getByRole('button', { name: 'Workspace' }).click();
  await page.waitForTimeout(1000);
  const btn = page.getByRole('button', { name: 'Generate Color' });
  if (!(await btn.count())) return { ok:false, reason:'button_missing' };

  const before = uiEventReq;
  await btn.first().evaluate((el) => {
    for (let i = 0; i < 20; i += 1) el.click();
  });
  await page.waitForTimeout(500);
  const afterBurst = uiEventReq;

  const statusText = async () => {
    const txt = await page.textContent('body');
    const m = txt && txt.match(/color_updated_#[0-9a-fA-F]{6}|loading|ready|send_failed/g);
    return m ? m[m.length - 1] : null;
  };
  const s0 = await statusText();
  const t0 = Date.now();
  let changed = false;
  while (Date.now() - t0 < 25000) {
    const s1 = await statusText();
    if (s1 && s1 !== s0 && s1.startsWith('color_updated_')) {
      changed = true;
      break;
    }
    await page.waitForTimeout(250);
  }

  await page.waitForTimeout(1000);
  const afterDrain = uiEventReq;
  return {
    ok: true,
    burstDelta: afterBurst - before,
    drainDelta: afterDrain - before,
    changed,
    disabledNow: await btn.isDisabled(),
    loadingNow: await btn.first().evaluate((el) => el.classList.contains('is-loading')),
  };
})(page);
```
